1. 什么是 Address（地址类型）？

概念定义

address 是 Solidity 中专门用于存储以太坊地址的数据类型。以太坊地址是一个 20 字节（160 位） 的十六进制字符串，代表区块链上的一个账户或合约。

示例 
// 地址示例（42个字符，包含0x前缀）
0x742d35Cc6634C0532925a3b844Bc9e7595f0bEb5

地址的两种类型

类型 定义 功能 使用场景
address 普通地址类型 只能查询余额，不能接收 ETH 普通账户、合约地址
address payable 可支付地址类型 可以查询余额，可以接收 ETH 收款地址、资金库、需要转账的地址
为什么需要区分？

Solidity 设计者为了安全性故意做了这个区分：

  1. 防止意外转账：编译器会强制检查，确保只有标记为 payable 的地址才能接收 ETH
  2. 明确意图：代码阅读者一眼就能看出哪些地址是用于收款的
  3. 避免资金损失：防止 ETH 被误转到不支持收款的合约或地址

地址类型完整代码示例

solidity 
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.24;

contract AddressExample {
    // ============ 地址变量声明 ============

    // 普通地址：只能存储地址，不能接收 ETH
    address public owner;
    address public userWallet;

    // payable 地址：可以接收 ETH
    address payable public treasury;      // 资金库
    address payable public paymentReceiver;  // 收款人

    // ============ 地址的基本操作 ============

    // 获取地址的 ETH 余额（任何地址都可以查余额）
    function getAddressBalance(address _addr) public view returns (uint256) {
        return _addr.balance;  // 返回 wei 单位的余额
    }

    // 获取合约自身的地址
    function getContractAddress() public view returns (address) {
        return address(this);  // this 指代当前合约实例
    }

    // 获取合约自身的 ETH 余额
    function getContractBalance() public view returns (uint256) {
        return address(this).balance;
    }

    // ============ 地址比较 ============

    // 检查两个地址是否相等
    function isSameAddress(address _a, address _b) public pure returns (bool) {
        return _a == _b;  // 地址比较是逐字节比较
    }

    // 检查地址是否为零地址
    function isZeroAddress(address _addr) public pure returns (bool) {
        // 零地址是默认值，表示"空"或"无效"
        return _addr == address(0);
    }

    // ============ 地址类型转换 ============

    // 普通地址 → payable 地址（需要显式转换）
    function toPayable(address _normalAddr) public pure returns (address payable) {
        // 使用 payable() 关键字进行转换
        return payable(_normalAddr);
    }

    // payable 地址 → 普通地址（隐式转换，自动完成）
    function fromPayable(address payable _payableAddr) public pure returns (address) {
        // 可以直接赋值，不需要显式转换
        return _payableAddr;
    }

    // ============ 转账操作（只有 payable 地址可以转账） ============

    // 向 payable 地址转账 ETH
    function sendETH(address payable _receiver, uint256 _amount) public {
        // transfer 方法：失败时自动 revert（回滚交易）
        _receiver.transfer(_amount);
    }
}
关键知识点总结
  1. 所有以太坊地址都是 20 字节，与类型无关
  2. address 和 address payable 可以互相转换：
    • address → address payable: 需要使用 payable() 显式转换
    • address payable → address: 可以直接赋值（隐式转换）
  3. 只有 address payable 可以调用 .transfer() 和 .send() 方法
  4. 任何地址都可以查询 .balance 属性

2. 什么是 msg.sender？

概念定义

msg.sender 是 Solidity 的全局变量，表示当前函数调用的直接发起者（一个地址）。

重要特性

solidity 
// 特性1: msg.sender 的值会根据调用上下文变化
// 特性2: 在构造函数中，msg.sender 是合约的部署者
// 特性3: 在普通函数中，msg.sender 是调用该函数的账户或合约
// 特性4: 如果合约A调用合约B，那么在合约B中 msg.sender 是合约A的地址

代码示例详解

solidity 
contract MsgSenderExample {
    // 记录不同阶段的 msg.sender
    address public deployer;      // 部署者
    address public lastCaller;    // 最后调用者

    // ============ 构造函数 ============

    constructor() {
        // 部署合约时，msg.sender 是执行部署交易的账户地址
        deployer = msg.sender;
    }

    // ============ 普通函数 ============

    // 任何人都可以调用，记录调用者
    function doSomething() public {
        // 此时 msg.sender 是调用 doSomething() 的账户
        lastCaller = msg.sender;
    }

    // 只有部署者可以调用
    function onlyDeployerCanCall() public view returns (string memory) {
        // require 检查：如果不是部署者，交易会被拒绝
        require(msg.sender == deployer, "Only deployer can call this");
        return "Hello, deployer!";
    }

    // 获取当前调用者的地址
    function whoIsCalling() public view returns (address) {
        return msg.sender;
    }
}

经典的 Ownable 权限控制模式

solidity 
// 经典的 Ownable（可拥有）模式
contract Ownable {
    address public owner;  // 记录合约所有者

    constructor() {
        // 部署时设置 owner 为部署者
        owner = msg.sender;
    }

    // 修饰符：限制只有 owner 可以调用
    modifier onlyOwner() {
        require(msg.sender == owner, "Not the owner");
        _;  // 这里会插入被修饰函数的代码
    }

    // 只有 owner 可以调用的函数
    function sensitiveFunction() public onlyOwner {
        // 敏感操作...
    }
}

3. 什么是 Event（事件）？

概念定义

Event（事件）是 Solidity 提供的一种日志记录机制。当合约执行过程中触发事件时，会将相关信息写入区块链的交易日志中。

事件的作用
┌─────────────────────────────────────────────────────────────┐ │ 事件的作用 │ ├─────────────────────────────────────────────────────────────┤ │ │ │ 1. 省 Gas │ │ - 存储数据到合约状态变量：贵（永久存储） │ │ - 记录事件到交易日志：便宜（临时存储） │ │ │ │ 2. 方便外部监听 │ │ - DApp 可以实时订阅事件 │ │ - 钱包可以显示交易详情 │ │ │ │ 3. 可追溯历史 │ │ - 可以查询合约的所有历史操作 │ │ - 方便审计和调试 │ │ │ └─────────────────────────────────────────────────────────────┘

事件定义与使用完整示例

solidity 
contract EventExample {
    // ============ 事件定义 ============

    // 基础事件：记录转账
    // indexed 表示这个参数可以被索引搜索（最多3个indexed参数）
    event Transfer(
        address indexed from,    // 转出地址（可索引）
        address indexed to,      // 转入地址（可索引）
        uint256 amount           // 金额（不可索引）
    );

    // 复杂事件：记录状态变更
    event StatusChanged(
        address indexed user,    // 谁触发的
        uint256 indexed oldStatus,  // 旧状态
        uint256 indexed newStatus,  // 新状态
        string reason            // 变更原因
    );

    // 无参数事件：简单标记
    event ContractPaused();
    event ContractUnpaused();

    // ============ 事件触发 ============

    function transfer(address _to, uint256 _amount) public {
        // ... 执行转账逻辑 ...

        // 触发事件：emit 关键字 + 事件名 + 参数
        emit Transfer(msg.sender, _to, _amount);
    }

    function changeStatus(uint256 _newStatus, string memory _reason) public {
        uint256 oldStatus = 0;  // 假设查询到旧状态

        // ... 执行状态变更 ...

        // 触发事件
        emit StatusChanged(msg.sender, oldStatus, _newStatus, _reason);
    }

    function pause() public {
        // ... 暂停逻辑 ...
        emit ContractPaused();  // 无参数事件
    }
}
indexed 关键字详解
// indexed 参数可以被外部快速搜索和过滤
// 一个事件最多有 3 个 indexed 参数

event Deposit(
    address indexed user,     // ✅ 可以搜索"某个用户的所有存款"
    uint256 indexed amount,   // ✅ 可以搜索"特定金额的所有存款"
    uint256 timestamp         // ❌ 没有 indexed，只能读取不能搜索
);

// 外部查询示例：
// - 查询 user = 0x123... 的所有 Deposit 事件 ✓
// - 查询 amount = 1000 的所有 Deposit 事件 ✓
// - 查询 timestamp > 1000000 的所有 Deposit 事件 ✗

4. 什么是 Constructor（构造函数）？

概念定义

Constructor（构造函数）是合约中一个特殊的函数，它在合约部署时自动执行一次，用于初始化合约的状态变量。

构造函数的特点

solidity 
// 特点1: 每个合约可以有 0 个或 1 个构造函数
// 特点2: 不写构造函数时，Solidity 自动生成一个空的
// 特点3: 构造函数只执行一次，之后无法再调用
// 特点4: 构造函数可以是 payable（用于工厂合约部署）
// 特点5: 构造函数中可以使用 msg.sender（部署者地址）

构造函数完整示例

solidity 
contract ConstructorExample {
    // ============ 状态变量 ============

    address public owner;           // 合约所有者
    address public manager;         // 管理员
    uint256 public creationTime;    // 创建时间
    string public name;             // 合约名称
    bool public initialized;        // 是否已初始化

    // ============ 构造函数 ============

    // 带参数的构造函数
    constructor(
        address _manager,      // 外部传入的管理员地址
        string memory _name    // 合约名称
    ) {
        // msg.sender: 部署此合约的账户地址
        owner = msg.sender;

        // 设置传入的参数
        manager = _manager;
        name = _name;

        // 记录创建时间
        creationTime = block.timestamp;

        // 标记已初始化
        initialized = true;
    }

    // ============ 使用场景 ============

    // 检查是否是 owner
    function isOwner(address _addr) public view returns (bool) {
        return _addr == owner;
    }
}

5. 什么是 Mapping（映射）？

概念定义

Mapping（映射）是 Solidity 中的键值对存储结构，类似于其他语言中的字典（Dictionary）或哈希表（Hash Map）。

映射的语法

solidity 
// 基本语法
mapping(KeyType => ValueType) public variableName;

// KeyType（键类型）：只能是基本类型（address, uint, bytes 等）
// ValueType（值类型）：可以是任何类型（包括数组、结构体、另一个映射）

映射完整示例

solidity 
contract MappingExample {
    // ============ 基础映射 ============

    // 地址 → bool：记录是否是成员
    // 用途：白名单、权限检查
    mapping(address => bool) public isMember;

    // 地址 → uint：记录余额
    // 用途：账户余额、积分、代币余额
    mapping(address => uint256) public balances;

    // ============ 嵌套映射 ============

    // 地址 → (地址 → uint)：记录对某个代币的授权额度
    // 用途：ERC20 的 approve 机制
    mapping(address => mapping(address => uint256)) public allowance;

    // 使用示例：
    // allowance[owner][spender] = 1000
    // 表示 owner 授权 spender 可以花费 1000 个代币

    // ============ 映射操作 ============

    // 添加/更新成员
    function addMember(address _user) public {
        isMember[_user] = true;  // 直接赋值，不存在则创建
    }

    // 查询成员
    function checkMember(address _user) public view returns (bool) {
        return isMember[_user];  // 如果不存在，返回默认值 false
    }

    // 增加余额
    function addBalance(address _user, uint256 _amount) public {
        balances[_user] += _amount;  // 直接累加
    }

    // 查询余额
    function getBalance(address _user) public view returns (uint256) {
        return balances[_user];  // 如果不存在，返回默认值 0
    }

    // 设置授权额度
    function approve(address _spender, uint256 _amount) public {
        // msg.sender 是授权方（owner）
        allowance[msg.sender][_spender] = _amount;
    }

    // 查询授权额度
    function getAllowance(address _owner, address _spender)
        public
        view
        returns (uint256)
    {
        return allowance[_owner][_spender];
    }
}

6. 什么是 Contract Type（合约类型）？

概念定义

在 Solidity 中，每个合约定义本身就是一种类型。就像 uint、string 是类型一样，SimpleToken、AddressDemo 这些合约名也是类型。

类比理解

类比 
// 类比理解：
// Java: UserService userService = new UserService();
//          ↑类型      ↑变量名

// Solidity: SimpleToken public token;
//              ↑类型      ↑变量名

合约类型的两种使用方式

solidity 
// 方式1: 引用已部署的合约（通过地址）
SimpleToken token = SimpleToken(0x123...);

// 方式2: 部署新合约
SimpleToken newToken = new SimpleToken(1000);

合约类型完整示例

solidity 
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.24;

// ============ 被引用的合约 ============

contract SimpleToken {
    string public name = "Simple Token";
    string public symbol = "ST";
    uint256 public totalSupply;
    mapping(address => uint256) public balanceOf;

    constructor(uint256 _initialSupply) {
        totalSupply = _initialSupply;
        balanceOf[msg.sender] = _initialSupply;
    }

    function transfer(address _to, uint256 _value) public returns (bool) {
        require(balanceOf[msg.sender] >= _value, "Insufficient balance");
        balanceOf[msg.sender] -= _value;
        balanceOf[_to] += _value;
        return true;
    }
}

// ============ 引用合约的合约 ============

import "./SimpleToken.sol";  // 导入合约定义

contract ContractTypeExample {
    // ============ 合约类型变量声明 ============

    // 声明一个 SimpleToken 类型的变量
    // 这就像说："我要存一个 SimpleToken 合约的引用"
    SimpleToken public token;

    // 也可以同时存地址
    address public tokenAddress;

    // ============ 设置合约引用 ============

    // 通过地址设置合约引用
    function setToken(address _tokenAddress) public {
        // 安全检查：不能是零地址
        require(_tokenAddress != address(0), "Invalid address");

        // 关键步骤：将 address 转换为合约类型
        // 这就像告诉编译器："相信我，这个地址上部署的是 SimpleToken"
        token = SimpleToken(_tokenAddress);

        // 同时保存地址
        tokenAddress = _tokenAddress;
    }

    // ============ 调用其他合约的函数 ============

    // 调用 token 合约的 name() 函数
    function getTokenName() public view returns (string memory) {
        // 先检查是否已设置
        require(address(token) != address(0), "Token not set");

        // 像调用本地函数一样调用其他合约的函数
        return token.name();
    }

    // 调用 token 合约的 balanceOf() 函数
    function getMyTokenBalance() public view returns (uint256) {
        // 查询当前调用者在 token 合约中的余额
        return token.balanceOf(msg.sender);
    }

    // 通过本合约转账 token
    function transferToken(address _to, uint256 _amount) public returns (bool) {
        // 注意：这里 msg.sender 是调用者
        // 但 token.transfer 中的 msg.sender 会变成本合约的地址！
        return token.transfer(_to, _amount);
    }

    // ============ 部署新合约 ============

    // 在本合约中部署新的 SimpleToken 合约
    function createNewToken(uint256 _initialSupply)
        public
        returns (address)
    {
        // 使用 new 关键字部署新合约
        // 这会创建一个新的合约实例，并返回合约地址
        SimpleToken newToken = new SimpleToken(_initialSupply);

        // 获取新合约的地址
        address newTokenAddress = address(newToken);

        // 可以自动设置为当前 token
        token = newToken;
        tokenAddress = newTokenAddress;

        return newTokenAddress;
    }

    // ============ 合约类型和地址的转换 ============

    // 合约类型 → 地址
    function getTokenAsAddress() public view returns (address) {
        return address(token);  // 直接转换
    }

    // 地址 → 合约类型
    function convertAddressToToken(address _addr)
        public
        pure
        returns (SimpleToken)
    {
        return SimpleToken(_addr);  // 显式转换
    }
}

7. 全局变量详解（msg / block / tx）

msg 系列：当前消息调用信息

solidity 
contract MsgVariables {
    // ============ msg.sender ============
    // 含义：当前函数调用的直接发送者（地址）
    // 使用场景：权限控制、记录操作人

    function getSender() public view returns (address) {
        return msg.sender;
    }

    // ============ msg.value ============
    // 含义：随本次调用发送的 ETH 数量（单位：wei）
    // 注意：函数必须标记为 payable 才能接收 ETH
    // 1 ETH = 10^18 wei

    function deposit() public payable {
        // msg.value 就是用户发送的 ETH 数量
        uint256 amount = msg.value;
        // ... 处理存款逻辑 ...
    }

    // ============ msg.data ============
    // 含义：完整的调用数据（calldata），包含函数选择器和参数
    // 使用场景：底层合约交互、代理合约

    function getCallData() public pure returns (bytes memory) {
        return msg.data;
    }

    // ============ msg.sig ============
    // 含义：函数调用的选择器（selector），即 msg.data 的前4字节
    // 计算方式：keccak256("functionName(paramTypes)") 的前4字节

    function getFunctionSelector() public pure returns (bytes4) {
        return msg.sig;
    }
    // 例如：调用 getSender() 时，msg.sig = 0x0a...（4字节）
}

block 系列：区块信息

solidity 
contract BlockVariables {
    // ============ block.timestamp ============
    // 含义：当前区块的时间戳（Unix 时间，单位：秒）
    // 用途：时间锁、限时活动、锁仓释放
    // 注意：可以被矿工操纵（大约 ±15 秒范围），不要用于高价值随机

    uint256 public unlockTime;

    function setUnlockTime(uint256 _delay) public {
        unlockTime = block.timestamp + _delay;  // 例如：当前时间 + 7天
    }

    function withdraw() public {
        require(block.timestamp >= unlockTime, "Not yet unlocked");
        // ... 执行提款 ...
    }

    // ============ block.number ============
    // 含义：当前区块的编号（高度）
    // 用途：投票期限、锁定期间
    // 比 timestamp 更精确，每个区块增加 1

    uint256 public startBlock;
    uint256 public constant VOTING_PERIOD = 100;  // 100 个区块

    function startVoting() public {
        startBlock = block.number;
    }

    function isVotingActive() public view returns (bool) {
        return block.number < startBlock + VOTING_PERIOD;
    }

    // ============ block.coinbase ============
    // 含义：挖出当前区块的矿工/验证者的地址
    // 用途：矿工奖励分配（很少直接使用）

❓常见问题与解答

1 npm run node 报错 `EACCES: permission denied 127.0.0.1:8545` ▼

问题描述：Windows 系统运行 npm run node 启动 Hardhat 节点时，报错端口权限被拒绝。

原因分析：端口 8545 被 Windows 系统保留或已被其他程序占用。

解决方案
  1. 检查端口占用：netstat -ano | findstr :8545
  2. 更换 Hardhat 端口：npx hardhat node --port 8546
  3. 重启 winnat 服务（以管理员身份运行 PowerShell）： 
    net stop winnat
net start winnat
2 `mapping(address => bool)` 是什么意思？如何使用？ ▼

mapping 是 Solidity 中的键值对存储结构，语法为 mapping(KeyType => ValueType)。

solidity 
// 定义：地址 → 布尔值的映射
mapping(address => bool) public isMember;

// 含义解释：
// - KeyType: address（键，用于查询）
// - ValueType: bool（值，存储的数据）
// - 整体含义：给定一个地址，查询它是否是成员（true/false）

// 使用示例：
function addMember(address _user) public {
    isMember[_user] = true;  // 设置键值对
}

function checkMember(address _user) public view returns (bool) {
    return isMember[_user];  // 查询键对应的值
    // 如果从未设置过，返回默认值 false
}
类比理解

就像 Excel 表格：A列是地址，B列是布尔值；或者 Python 字典：{"0x123...": true, "0x456...": false}

3 address 和 address payable 的本质区别是什么？ ▼
技术本质

两者都是 20 字节的以太坊地址，区别在于编译器允许的操作不同。

solidity 
// address：只能查询，不能转账
address normalAddr = 0x123...;
normalAddr.balance;              // ✅ 可以查询余额
// normalAddr.transfer(1 ether); // ❌ 编译错误！普通地址不能转账

// address payable：可以查询，也可以转账
address payable payableAddr = payable(0x123...);
payableAddr.balance;             // ✅ 可以查询余额
payableAddr.transfer(1 ether);   // ✅ 可以转账
设计原因

Solidity 设计者为了安全，强制开发者显式标记哪些地址是"可收款的"，防止意外转账。

转换规则

// address → address payable（需要显式转换）
address payable p = payable(normalAddr);

// address payable → address（自动转换）
address n = payableAddr;  // 直接赋值即可
4 如何安全地接收和发送 ETH？ ▼

接收 ETH

// 方式1：receive 回调函数
receive() external payable {
    balances[msg.sender] += msg.value;
}

// 方式2：payable 函数
function deposit() external payable {
    require(msg.value > 0, "Must send ETH");
    balances[msg.sender] += msg.value;
}

发送 ETH（安全模式）

function withdraw(uint256 _amount) public {
    // 1. 检查
    require(balances[msg.sender] >= _amount, "Insufficient");

    // 2. 更新（先扣款！）
    balances[msg.sender] -= _amount;

    // 3. 转账（后转账！）
    payable(msg.sender).transfer(_amount);
}
为什么要"先扣款后转账"？

防止"重入攻击"——恶意合约在收到 ETH 时自动再次调用 withdraw，如果先转账，余额还没扣，就可以多次取款。

5 合约类型详解：如何引用其他合约？ ▼ 
// 步骤1：导入合约定义
import "./SimpleToken.sol";

// 步骤2：声明合约类型变量
SimpleToken public token;  // token 是变量名，SimpleToken 是类型

// 步骤3：通过地址设置引用
function setToken(address _addr) public {
    token = SimpleToken(_addr);  // 地址 → 合约类型转换
}

// 步骤4：调用合约函数
function getName() public view returns (string memory) {
    return token.name();  // 像调用本地函数一样
}
核心理解
  • SimpleToken = 合约的"设计图纸"（类型）
  • token = 具体的"房子实例"（变量）
  • SimpleToken(_addr) = 告诉编译器"这个地址上是 SimpleToken"

模板1：完整的 Ownable 权限控制合约
提供基本的权限控制功能，只有一个 owner 可以执行敏感操作
solidity 
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.24;

contract Ownable {
    address private _owner;

    event OwnershipTransferred(
        address indexed previousOwner,
        address indexed newOwner
    );

    modifier onlyOwner() {
        require(msg.sender == _owner, "Ownable: caller is not the owner");
        _;
    }

    constructor() {
        _owner = msg.sender;
        emit OwnershipTransferred(address(0), msg.sender);
    }

    function owner() public view returns (address) {
        return _owner;
    }

    function transferOwnership(address newOwner) public onlyOwner {
        require(newOwner != address(0), "Ownable: new owner is the zero address");

        address oldOwner = _owner;
        _owner = newOwner;

        emit OwnershipTransferred(oldOwner, newOwner);
    }
}
模板2：安全的 ETH 存取合约
使用 Checks-Effects-Interactions 模式，防止重入攻击
solidity 
contract ETHBank {
    mapping(address => uint256) private _balances;
    bool private _locked;

    event Deposit(address indexed account, uint256 amount);
    event Withdrawal(address indexed account, uint256 amount);

    modifier nonReentrant() {
        require(!_locked, "Reentrant call");
        _locked = true;
        _;
        _locked = false;
    }

    receive() external payable {
        _deposit();
    }

    function deposit() external payable {
        _deposit();
    }

    function _deposit() internal {
        require(msg.value > 0, "Must send ETH");
        _balances[msg.sender] += msg.value;
        emit Deposit(msg.sender, msg.value);
    }

    function withdraw(uint256 amount) external nonReentrant {
        // Checks
        require(amount > 0, "Amount must be positive");
        require(_balances[msg.sender] >= amount, "Insufficient balance");

        // Effects（先扣减）
        _balances[msg.sender] -= amount;

        // Interactions（后转账）
        (bool success, ) = payable(msg.sender).call{value: amount}("");
        require(success, "Transfer failed");

        emit Withdrawal(msg.sender, amount);
    }

    function balanceOf(address account) public view returns (uint256) {
        return _balances[account];
    }
}
模板3：合约交互 - TokenManager
管理代币合约引用，展示如何调用其他合约的函数
solidity 
interface IERC20 {
    function name() external view returns (string memory);
    function symbol() external view returns (string memory);
    function totalSupply() external view returns (uint256);
    function balanceOf(address account) external view returns (uint256);
    function transfer(address to, uint256 amount) external returns (bool);
    function transferFrom(address from, address to, uint256 amount) external returns (bool);
}

contract TokenManager {
    IERC20 public token;
    address public tokenAddress;

    event TokenSet(address indexed tokenAddress);

    function setToken(address _tokenAddress) external {
        require(_tokenAddress != address(0), "Invalid token address");

        // 地址 → 合约类型转换
        token = IERC20(_tokenAddress);
        tokenAddress = _tokenAddress;

        emit TokenSet(_tokenAddress);
    }

    modifier tokenSet() {
        require(tokenAddress != address(0), "Token not set");
        _;
    }

    function getTokenName() external view tokenSet returns (string memory) {
        return token.name();
    }

    function getTokenBalance(address account)
        external
        view
        tokenSet
        returns (uint256)
    {
        return token.balanceOf(account);
    }
}

🛡️安全最佳实践

✅ 应该做的

// 1. 使用 msg.sender 做权限控制
require(msg.sender == owner);

// 2. 检查零地址
require(_addr != address(0));

// 3. Checks-Effects-Interactions
function withdraw(uint256 amount) public {
    require(balances[msg.sender] >= amount); // Checks
    balances[msg.sender] -= amount;         // Effects
    payable(msg.sender).transfer(amount);      // Interactions
}

// 4. 使用 transfer 或带保护的 call
payable(addr).transfer(amount);

❌ 不应该做的

// 1. 不要用 tx.origin
require(tx.origin == owner);  // 危险！

// 2. 不要先转账后更新
payable(msg.sender).transfer(amount);  // ❌
balances[msg.sender] -= amount;          // 可能被重入

// 3. 不要用 block.timestamp 做随机
uint256 random = block.timestamp % 10;  // 可被操控

// 4. 不要忽略 send 的返回值
someAddress.send(amount);  // 可能失败但继续执行
