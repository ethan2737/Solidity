## 第一部分：接口（Interface）

### 1.1 概念介绍

接口（Interface）是 Solidity 中用于定义合约标准的重要机制。接口只定义函数签名，不包含实现代码，不能有状态变量。通过接口，可以实现：
- 代码复用
- 标准化设计（如 ERC20、ERC721）
- 合约间解耦

### 1.2 接口使用方法

1. 使用 `interface` 关键字定义接口
2. 接口中的函数必须是 `external` 可见性
3. 接口不能包含函数实现
4. 接口不能有状态变量
5. 合约使用 `is` 关键字实现接口
6. 实现接口的合约必须实现所有函数，使用 `override` 关键字

### 1.3 接口代码示例

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.24;

/**
 * @title IERC20Like
 * @notice 类似 ERC20 的接口定义
 * @dev 接口定义：定义合约必须实现的函数签名
 */
interface IERC20Like {
    // 接口函数必须是 external 可见性
    function totalSupply() external view returns (uint256);
    function balanceOf(address account) external view returns (uint256);
    function transfer(address to, uint256 amount) external returns (bool);
    function transferFrom(address from, address to, uint256 amount) external returns (bool);
    function approve(address spender, uint256 amount) external returns (bool);
    function allowance(address owner, address spender) external view returns (uint256);

    // 接口可以包含事件定义
    event Transfer(address indexed from, address indexed to, uint256 value);
    event Approval(address indexed owner, address indexed spender, uint256 value);
}

/**
 * @title SimpleToken
 * @notice 实现 IERC20Like 接口的简单代币合约
 * @dev 接口实现：合约通过实现接口来遵循标准
 */
contract SimpleToken is IERC20Like {
    mapping(address => uint256) private _balances;
    mapping(address => mapping(address => uint256)) private _allowances;
    uint256 private _totalSupply;

    // 实现接口函数必须使用 override 关键字
    function totalSupply() external view override returns (uint256) {
        return _totalSupply;
    }

    function balanceOf(address account) external view override returns (uint256) {
        return _balances[account];
    }

    function transfer(address to, uint256 amount) external override returns (bool) {
        require(_balances[msg.sender] >= amount, "Insufficient balance");
        _balances[msg.sender] -= amount;
        _balances[to] += amount;
        emit Transfer(msg.sender, to, amount);
        return true;
    }

    // ... 其他函数实现
}

/**
 * @title IStorable
 * @notice 存储接口
 */
interface IStorable {
    function store(uint256 value) external;
    function retrieve() external view returns (uint256);
}

/**
 * @title IWithdrawable
 * @notice 可提取接口
 */
interface IWithdrawable {
    function withdraw(uint256 amount) external;
    function deposit() external payable;
}

/**
 * @title WalletContract
 * @notice 实现多个接口的钱包合约
 * @dev 一个合约可以实现多个接口
 */
contract WalletContract is IStorable, IWithdrawable {
    uint256 private storedValue;
    mapping(address => uint256) private balances;

    function store(uint256 value) external override {
        storedValue = value;
    }

    function retrieve() external view override returns (uint256) {
        return storedValue;
    }

    function deposit() external payable override {
        balances[msg.sender] += msg.value;
    }

    function withdraw(uint256 amount) external override {
        require(balances[msg.sender] >= amount, "Insufficient balance");
        balances[msg.sender] -= amount;
        payable(msg.sender).transfer(amount);
    }
}

/**
 * @title InterfaceUser
 * @notice 演示如何使用接口与其他合约交互
 * @dev 通过接口调用其他合约，实现合约间交互
 */
contract InterfaceUser {
    // 通过接口调用存储合约的 store 函数
    function storeViaInterface(address storageContract, uint256 value) external {
        IStorable(storageContract).store(value);
    }

    // 通过接口读取存储合约的值
    function retrieveViaInterface(address storageContract) external view returns (uint256) {
        return IStorable(storageContract).retrieve();
    }

    // 通过接口转账代币
    function transferToken(address token, address to, uint256 amount) external returns (bool) {
        return IERC20Like(token).transfer(to, amount);
    }

    // 检查合约是否实现了接口
    function checkInterface(address contractAddress) external view returns (bool) {
        try IStorable(contractAddress).retrieve() returns (uint256) {
            return true;
        } catch {
            return false;
        }
    }
}
```

### 1.4 接口使用注意事项

1. **接口函数必须是 external**：接口中定义的所有函数必须是 `external` 可见性
2. **不能包含实现**：接口只能定义函数签名，不能包含函数体
3. **不能有状态变量**：接口不能定义任何状态变量
4. **必须实现所有函数**：实现接口的合约必须实现所有函数，否则会编译错误
5. **可继承其他接口**：接口可以继承其他接口
6. **命名规范**：接口名通常以 "I" 开头，如 `IERC20`、`IStorable`

### 1.5 接口核心要点总结

- **接口 = 外部约束**：定义合约必须有哪些方法，规范合约标准
- **接口约其他合约**：通过接口与已部署合约交互，直接使用接口名传入合约地址调用方法
- **接口 vs 实现**：接口定义"做什么"，实现合约定义"怎么做"
- **try/catch 作用**：捕获外部合约调用的异常（如目标合约未实现接口方法）

## 第二部分：继承（Inheritance）

### 2.1 概念介绍

继承（Inheritance）是 Solidity 中实现代码复用的重要机制。通过继承，子合约可以复用父合约的状态变量、函数和修饰符。Solidity 支持单继承和多继承。

### 2.2 继承使用方法

1. 使用 `is` 关键字实现继承
2. **单继承**：一个合约继承一个合约
3. **多继承**：一个合约继承多个合约，用逗号分隔
4. **函数重写**：使用 `override` 关键字重写父合约函数
5. **构造函数继承**：在子合约构造函数中调用父合约构造函数
6. **super 关键字**：调用父合约函数

### 2.3 继承代码示例

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.24;

/**
 * @title Ownable
 * @notice 所有权管理基础合约
 * @dev 用于演示基础合约的继承
 */
contract Ownable {
    address public owner;

    event OwnershipTransferred(address indexed previousOwner, address indexed newOwner);

    constructor() {
        owner = msg.sender;
        emit OwnershipTransferred(address(0), msg.sender);
    }

    // 修饰符：限制只有所有者可以调用
    modifier onlyOwner() {
        require(msg.sender == owner, "Ownable: caller is not the owner");
        _;
    }

    function transferOwnership(address newOwner) public virtual onlyOwner {
        require(newOwner != address(0), "Ownable: new owner is the zero address");
        address oldOwner = owner;
        owner = newOwner;
        emit OwnershipTransferred(oldOwner, newOwner);
    }
}

/**
 * @title Pausable
 * @notice 可暂停功能基础合约
 */
contract Pausable {
    bool public paused;

    event Paused(address account);
    event Unpaused(address account);

    modifier whenNotPaused() {
        require(!paused, "Pausable: paused");
        _;
    }

    modifier whenPaused() {
        require(paused, "Pausable: not paused");
        _;
    }

    function pause() public virtual {
        paused = true;
        emit Paused(msg.sender);
    }

    function unpause() public virtual {
        paused = false;
        emit Unpaused(msg.sender);
    }
}

/**
 * @title StorageBase
 * @notice 存储基础合约
 */
contract StorageBase {
    uint256 public storedValue;

    event ValueStored(uint256 oldValue, uint256 newValue);

    function store(uint256 value) public virtual {
        uint256 oldValue = storedValue;
        storedValue = value;
        emit ValueStored(oldValue, value);
    }

    function retrieve() public view virtual returns (uint256) {
        return storedValue;
    }
}

/**
 * @title OwnableStorage
 * @notice 继承 Ownable 的存储合约（单继承）
 * @dev 单继承示例
 */
contract OwnableStorage is Ownable {
    uint256 private _value;

    event ValueChanged(uint256 oldValue, uint256 newValue);

    function setValue(uint256 newValue) public onlyOwner {
        uint256 oldValue = _value;
        _value = newValue;
        emit ValueChanged(oldValue, newValue);
    }

    function getValue() public view returns (uint256) {
        return _value;
    }
}

/**
 * @title OwnablePausableStorage
 * @notice 继承多个合约的存储合约（多继承）
 * @dev 多继承示例
 */
contract OwnablePausableStorage is Ownable, Pausable, StorageBase {
    /**
     * @notice 构造函数，调用父合约构造函数
     * @param initialValue 初始值
     */
    constructor(uint256 initialValue) Ownable() {
        storedValue = initialValue;
    }

    /**
     * @notice 重写 store 函数，添加暂停检查
     * @dev 使用 override 关键字重写父合约函数
     */
    function store(uint256 value) public override whenNotPaused {
        super.store(value);  // 调用父合约函数
    }

    function pause() public override onlyOwner {
        super.pause();
    }

    function unpause() public override onlyOwner {
        super.unpause();
    }
}

/**
 * @title AdvancedStorage
 * @notice 继承并重写函数的存储合约
 * @dev 函数重写示例
 */
contract AdvancedStorage is StorageBase {
    uint256 public maxValue;
    uint256 public minValue;

    constructor(uint256 _maxValue, uint256 _minValue) {
        maxValue = _maxValue;
        minValue = _minValue;
    }

    /**
     * @notice 重写 store 函数，添加范围检查
     */
    function store(uint256 value) public override {
        require(value >= minValue && value <= maxValue, "Value out of range");
        super.store(value);
    }

    function retrieve() public view override returns (uint256) {
        uint256 baseValue = super.retrieve();
        return baseValue;
    }
}

/**
 * @title AbstractToken
 * @notice 抽象合约，定义接口但不完全实现
 * @dev 抽象合约可以包含部分实现和状态变量
 */
abstract contract AbstractToken {
    mapping(address => uint256) public balances;

    // 抽象函数：没有实现，需要子类实现
    function transfer(address to, uint256 amount) public virtual returns (bool);

    // 具体函数：有默认实现
    function balanceOf(address account) public view returns (uint256) {
        return balances[account];
    }
}

/**
 * @title ConcreteToken
 * @notice 实现抽象合约的具体合约
 */
contract ConcreteToken is AbstractToken {
    function transfer(address to, uint256 amount) public override returns (bool) {
        require(balances[msg.sender] >= amount, "Insufficient balance");
        balances[msg.sender] -= amount;
        balances[to] += amount;
        return true;
    }

    function mint(address to, uint256 amount) public {
        balances[to] += amount;
    }
}

/**
 * @title FinalContract
 * @notice 继承链示例
 * @dev 多级继承：BaseContract -> MiddleContract -> FinalContract
 */
contract BaseContract {
    uint256 public baseValue;

    function setBaseValue(uint256 value) public virtual {
        baseValue = value;
    }
}

contract MiddleContract is BaseContract {
    uint256 public middleValue;

    function setBaseValue(uint256 value) public virtual override {
        super.setBaseValue(value);  // 调用父合约函数
        middleValue = value * 2;
    }
}

contract FinalContract is MiddleContract {
    uint256 public finalValue;

    function setBaseValue(uint256 value) public override {
        super.setBaseValue(value);
        finalValue = value * 3;
    }
}
```

### 2.4 继承使用注意事项

1. **函数重写必须使用 override**：重写父合约函数必须使用 `override` 关键字
2. **构造函数调用**：子合约构造函数必须调用父合约构造函数
3. **多继承顺序**：多继承时要注意父合约的顺序，避免函数冲突
4. **super 关键字**：使用 `super` 调用父合约函数，可以调用整个继承链的函数
5. **可见性**：子合约不能降低父合约函数的可见性
6. **抽象合约**：使用 `abstract` 关键字定义抽象合约，抽象合约可以包含未实现的函数

### 2.5 接口、继承、抽象合约概念对比

#### 接口（Interface）
- 通过 `interface` 关键字定义
- 仅定义函数签名，不写具体实现
- 不能包含变量、构造函数
- 通过 `is` 关键字实现，**必须使用 `override`** 覆盖所有方法
- 一对一或一对多（多接口）
- 用于与其他合约交互（跨合约调用）
- 使用 `try/catch` 捕获外部调用异常

#### 继承（Inheritance）
- 两个独立合约之间的交互关系
- 可以有变量、构造函数、方法
- 通过 `is` 关键字实现
- 一对一或一对多（多继承）
- 可选使用 `override` 重写函数
- 可使用 `super` 关键字调用父合约方法

#### 抽象合约（Abstract Contract）
- 通过 `abstract` 关键字定义
- 可以包含已实现的函数 + 未实现的函数
- 可以有状态变量、构造函数
- 子合约**必须**实现所有抽象方法
- 目的：定义通用逻辑（复用）+ 强制规范（抽象方法）

#### 继承链
多层继承形成链式结构：
```
Animal (抽象) → Mammal (抽象) → Dog (具体实现)
```
子合约需要实现继承链上所有的抽象方法。

## 第三部分：库（Library）

### 3.1 概念介绍

库（Library）是 Solidity 中用于封装通用功能的机制。库函数通过 `delegatecall` 方式调用，可以实现代码复用和 Gas 优化。库不能有状态变量，不能接收 Ether。

### 3.2 库使用方法

1. 使用 `library` 关键字定义库
2. 库函数必须是 `internal` 或 `public`
3. 不能有状态变量
4. 不能接收 Ether
5. 使用 `using ... for` 语法将库绑定到类型
6. 也可以直接调用库函数

### 3.3 库代码示例

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.24;

/**
 * @title SafeMath
 * @notice 安全的数学运算库
 * @dev 虽然 Solidity 0.8+ 自带溢出检查，但这里演示库的用法
 */
library SafeMath {
    /**
     * @notice 安全的加法
     * @param a 第一个数
     * @param b 第二个数
     * @return c 结果
     */
    function add(uint256 a, uint256 b) internal pure returns (uint256) {
        uint256 c = a + b;
        require(c >= a, "SafeMath: addition overflow");
        return c;
    }

    /**
     * @notice 安全的减法
     */
    function sub(uint256 a, uint256 b) internal pure returns (uint256) {
        require(b <= a, "SafeMath: subtraction overflow");
        return a - b;
    }

    /**
     * @notice 安全的乘法
     */
    function mul(uint256 a, uint256 b) internal pure returns (uint256) {
        if (a == 0) {
            return 0;
        }
        uint256 c = a * b;
        require(c / a == b, "SafeMath: multiplication overflow");
        return c;
    }

    /**
     * @notice 安全的除法
     */
    function div(uint256 a, uint256 b) internal pure returns (uint256) {
        require(b > 0, "SafeMath: division by zero");
        return a / b;
    }
}

/**
 * @title ArrayUtils
 * @notice 数组工具库
 */
library ArrayUtils {
    /**
     * @notice 查找数组中元素的索引
     * @param array 数组
     * @param value 要查找的值
     * @return 索引
     */
    function indexOf(uint256[] storage array, uint256 value) internal view returns (uint256) {
        for (uint256 i = 0; i < array.length; i++) {
            if (array[i] == value) {
                return i;
            }
        }
        revert("Value not found");
    }

    /**
     * @notice 移除数组中的元素
     */
    function remove(uint256[] storage array, uint256 index) internal {
        require(index < array.length, "Index out of bounds");
        for (uint256 i = index; i < array.length - 1; i++) {
            array[i] = array[i + 1];
        }
        array.pop();
    }

    /**
     * @notice 检查数组是否包含元素
     */
    function contains(uint256[] storage array, uint256 value) internal view returns (bool) {
        for (uint256 i = 0; i < array.length; i++) {
            if (array[i] == value) {
                return true;
            }
        }
        return false;
    }
}

/**
 * @title StringUtils
 * @notice 字符串工具库
 */
library StringUtils {
    /**
     * @notice 比较两个字符串是否相等
     */
    function equal(string memory a, string memory b) internal pure returns (bool) {
        return keccak256(abi.encodePacked(a)) == keccak256(abi.encodePacked(b));
    }

    /**
     * @notice 连接两个字符串
     */
    function concat(string memory a, string memory b) internal pure returns (string memory) {
        return string(abi.encodePacked(a, b));
    }
}

/**
 * @title AddressUtils
 * @notice 地址工具库
 */
library AddressUtils {
    /**
     * @notice 检查地址是否为合约
     * @dev 使用 assembly 读取合约代码大小
     */
    function isContract(address account) internal view returns (bool) {
        uint256 size;
        assembly {
            size := extcodesize(account)
        }
        return size > 0;
    }

    /**
     * @notice 检查地址是否为零地址
     */
    function isZero(address account) internal pure returns (bool) {
        return account == address(0);
    }
}

/**
 * @title MathContract
 * @notice 使用 SafeMath 库的合约
 */
contract MathContract {
    // 使用 using ... for 语法将 SafeMath 绑定到 uint256 类型
    using SafeMath for uint256;

    uint256 public value;

    function add(uint256 a, uint256 b) public pure returns (uint256) {
        // 使用 using ... for 语法调用库函数
        return a.add(b);
    }

    function subtract(uint256 a, uint256 b) public pure returns (uint256) {
        return a.sub(b);
    }

    function setValue(uint256 newValue) public {
        // 也可以直接调用库函数
        value = SafeMath.add(value, newValue);
    }
}

/**
 * @title ArrayContract
 * @notice 使用 ArrayUtils 库的合约
 */
contract ArrayContract {
    using ArrayUtils for uint256[];

    uint256[] public numbers;

    function addNumber(uint256 number) public {
        numbers.push(number);
    }

    function removeNumber(uint256 number) public {
        uint256 index = numbers.indexOf(number);
        numbers.remove(index);
    }

    function containsNumber(uint256 number) public view returns (bool) {
        return numbers.contains(number);
    }
}

/**
 * @title AddressContract
 * @notice 使用 AddressUtils 库的合约
 */
contract AddressContract {
    using AddressUtils for address;

    address public storedAddress;

    function setAddress(address newAddress) public {
        require(!newAddress.isZero(), "Cannot set zero address");
        storedAddress = newAddress;
    }

    function checkIfContract() public view returns (bool) {
        return storedAddress.isContract();
    }
}
```

### 3.4 库使用注意事项

1. **函数可见性**：库函数只能是 `internal` 或 `public`
2. **不能有状态变量**：库不能定义任何状态变量
3. **不能接收 Ether**：库不能有 `receive` 函数或 `payable` 函数
4. **using ... for 语法**：将库绑定到类型后，可以像类型的方法一样调用库函数
5. **直接调用**：也可以直接通过 `LibraryName.functionName()` 调用
6. **Gas 优化**：库函数调用比合约函数调用更省 Gas

### 3.5 库核心要点总结

- **库 = 操作封装**：通过 `library` 关键字定义，是一类操作的高度集合
- **using 库名 for 类型**：将库绑定到指定类型，简化调用方式
- **点操作调用**：绑定后，对应类型的数据可以直接通过 `.` 调用库方法
- **返回值必须一致**：调用库方法时，返回值类型必须与库中方法返回值一致，否则报错
- **无状态**：库不能存储状态，仅包含纯逻辑函数
- **Gas 优化**：库函数通过 `delegatecall` 调用，比合约函数更省 Gas

## 第四部分：Event 事件

### 4.1 概念介绍

Event（事件）是 Solidity 中用于记录合约状态变化的机制。事件数据存储在交易日志中，不占用存储空间，Gas 成本低。事件是前端监听和日志审计的重要组成部分。

### 4.2 Event 使用方法

1. 使用 `event` 关键字定义事件
2. 使用 `emit` 关键字触发事件
3. 使用 `indexed` 关键字标记可过滤参数（最多 3 个）
4. 匿名事件使用 `anonymous` 关键字，更省 Gas

### 4.3 Event 代码示例

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.24;

/**
 * @title EventDemo
 * @notice 演示 Solidity Event 的基础用法
 */
contract EventDemo {
    address public owner;
    uint256 public totalSupply;
    mapping(address => uint256) public balances;

    /**
     * @notice 基础事件示例
     * @dev indexed 参数可以用于过滤，最多 3 个
     * @param from 发送者地址（indexed，可用于过滤）
     * @param to 接收者地址（indexed，可用于过滤）
     * @param amount 转账金额
     */
    event Transfer(
        address indexed from,
        address indexed to,
        uint256 amount
    );

    /**
     * @notice 带多个参数的事件
     */
    event BalanceUpdated(
        address indexed account,
        uint256 oldBalance,
        uint256 newBalance,
        uint256 timestamp
    );

    /**
     * @notice 所有权转移事件
     */
    event OwnershipTransferred(
        address indexed previousOwner,
        address indexed newOwner
    );

    /**
     * @notice 代币铸造事件
     */
    event Mint(
        address indexed to,
        uint256 amount
    );

    /**
     * @notice 代币销毁事件
     */
    event Burn(
        address indexed from,
        uint256 amount
    );

    /**
     * @notice 授权事件
     */
    event Approval(
        address indexed owner,
        address indexed spender,
        uint256 amount
    );

    constructor(uint256 _initialSupply) {
        owner = msg.sender;
        totalSupply = _initialSupply;
        balances[msg.sender] = _initialSupply;

        // 触发初始事件
        emit Transfer(address(0), msg.sender, _initialSupply);
    }

    /**
     * @notice 转账函数，触发 Transfer 事件
     */
    function transfer(address to, uint256 amount) public {
        require(to != address(0), "Cannot transfer to zero address");
        require(balances[msg.sender] >= amount, "Insufficient balance");

        uint256 oldBalanceFrom = balances[msg.sender];
        uint256 oldBalanceTo = balances[to];

        balances[msg.sender] -= amount;
        balances[to] += amount;

        // 触发 Transfer 事件（事件是前端监听、日志审计的重要组成部分）
        emit Transfer(msg.sender, to, amount);

        // 触发 BalanceUpdated 事件
        emit BalanceUpdated(
            msg.sender,
            oldBalanceFrom,
            balances[msg.sender],
            block.timestamp
        );
    }

    /**
     * @notice 铸造代币，触发 Mint 事件
     */
    function mint(address to, uint256 amount) public {
        require(msg.sender == owner, "Only owner can mint");

        totalSupply += amount;
        balances[to] += amount;

        // 触发 Mint 事件（事件是前端监听、日志审计的重要组成部分）
        emit Mint(to, amount);
        emit Transfer(address(0), to, amount);
    }

    /**
     * @notice 销毁代币，触发 Burn 事件
     */
    function burn(uint256 amount) public {
        require(balances[msg.sender] >= amount, "Insufficient balance");

        balances[msg.sender] -= amount;
        totalSupply -= amount;

        // 触发 Burn 事件（事件是前端监听、日志审计的重要组成部分）
        emit Burn(msg.sender, amount);
        emit Transfer(msg.sender, address(0), amount);
    }

    /**
     * @notice 转移所有权，触发 OwnershipTransferred 事件
     */
    function transferOwnership(address newOwner) public {
        require(msg.sender == owner, "Only owner can transfer ownership");

        address oldOwner = owner;
        owner = newOwner;

        // 触发 OwnershipTransferred 事件
        emit OwnershipTransferred(oldOwner, newOwner);
    }

    /**
     * @notice 授权函数，触发 Approval 事件
     */
    mapping(address => mapping(address => uint256)) public allowances;

    function approve(address spender, uint256 amount) public returns (bool) {
        allowances[msg.sender][spender] = amount;

        // 触发 Approval 事件
        emit Approval(msg.sender, spender, amount);

        return true;
    }
}
```

### 4.5 Event 使用注意事项

1. **indexed 参数**：
   - 最多 3 个 `indexed` 参数
   - `indexed` 参数存储在日志的主题（topics）中，可以用于过滤
   - 非 `indexed` 参数存储在日志的数据（data）中，不能直接过滤

2. **匿名事件**：
   - 使用 `anonymous` 关键字声明
   - 更省 Gas，因为不存储事件签名
   - 无法通过事件签名过滤，必须使用完整的事件参数过滤

3. **事件是前端监听的重要组成部分**：
   - 前端可以通过监听事件实时更新 UI
   - 事件日志可以用于审计和追踪

4. **事件与存储的区别**：
   - 事件存储在交易日志中，不占用存储空间，Gas 成本低
   - 事件适合记录历史数据
   - Storage 适合存储当前状态

5. **Etherscan 查看**：
   - 部署到测试网后，可以在 Etherscan 查看事件日志
   - 事件是前端监听、日志审计的重要组成部分

## 最佳实践总结

### 接口最佳实践

1. **命名规范**：接口名以 "I" 开头，如 `IERC20`
2. **最小化接口**：只定义必要的函数
3. **使用接口进行合约交互**：通过接口调用其他合约，实现解耦

### 继承最佳实践

1. **使用继承实现代码复用**：将通用逻辑放在父合约中
2. **合理使用多继承**：注意继承顺序和函数冲突
3. **使用 override 关键字**：明确标识重写函数

### 库最佳实践

1. **封装通用函数**：将常用的工具函数放在库中
2. **使用 using ... for 语法**：让代码更简洁
3. **Gas 优化**：库函数调用更省 Gas

### Event 最佳实践

1. **使用事件记录重要状态变化**：如转账、授权等
2. **合理使用 indexed 参数**：最多 3 个，用于过滤
3. **使用匿名事件节省 Gas**：对于频繁触发的事件
4. **事件是前端监听的重要组成部分**：设计好事件便于前端实时更新
